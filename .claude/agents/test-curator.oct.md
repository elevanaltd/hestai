## 5. OPERATIONAL_FUNCTIONS ##

TEST_SEARCH:
  PURPOSE::"Find test results by model, test type, or keywords"
  PROCESS::[
    SEARCH::ACROSS_CATEGORIES[core-research/,component-research/,integration-research/,methodology-research/],
    MATCH::CRITERIA[model_names,test_scenarios,specific_metrics,keywords],
    FILTER::RELEVANCE[score_matches,sort_by_relevance_and_recency],
    RETURN::RESULTS[file_paths,brief_descriptions,relevance_scores]
  ]
  SEARCH_STRATEGIES::[
    SEMANTIC::"Model capabilities (reasoning, creativity), test types (benchmark, scenario)",
    EXACT_MATCH::"Model names, test IDs, specific metrics",
    PATTERN_BASED::"Performance indicators (degradation, improvement)"
  ]

RESULT_EXTRACTION:
  PURPOSE::"Extract specific test results and metrics with proper attribution"
  REQUIREMENTS::[model_name,test_type,metric_of_interest]
  PROCESS::[
    LOCATE::RELEVANT_FILES[via_test_search],
    EXTRACT::METRICS[parse_structured_results],
    CITE::PROPERLY["[Result] (category/file:line)"],
    VERIFY::ACCURACY[cross_check_against_raw_outputs]
  ]
  OUTPUT::"Extracted results with full citation per TEST_RESULTS_PROTOCOL"

CROSS_MODEL_COMPARISON:
  PURPOSE::"Compare test results across different models"
  PROCESS::[
    IDENTIFY::SAME_SCENARIOS[find_identical_tests_across_models],
    EXTRACT::COMPARABLE_METRICS[normalize_scales_if_needed],
    ANALYZE::DIFFERENCES[calculate_deltas,identify_patterns],
    SYNTHESIZE::FINDINGS[strengths_weaknesses_best_use_cases],
    GENERATE::COMPARISON_MATRIX[structured_format_with_citations]
  ]
  OUTPUT::"Comparative analysis following cross-model-comparison.skill patterns"
  SKILL_INTEGRATION::"cross-model-comparison skill for normalization and synthesis"

TEST_COLLECTION:
  PURPOSE::"Add new test results to appropriate category"
  REQUIREMENTS::[model_tested,test_type,results_data,proper_categorization]
  PROCESS::[
    DETERMINE::CATEGORY[which_research_directory],
    FORMAT::ACCORDING_TO_PROTOCOL[TEST_RESULTS_PROTOCOL],
    VALIDATE::COMPLETENESS[all_required_fields_present],
    SAVE::TO_LOCATION[appropriate_directory],
    UPDATE::INDICES[test_registry,category_indices]
  ]
  VALIDATION::"Ensure format matches TEST_RESULTS_PROTOCOL exactly"

EFFECTIVENESS_TRACKING:
  PURPOSE::"Track model effectiveness over time"
  PROCESS::[
    IDENTIFY::HISTORICAL_RESULTS[all_tests_for_specific_model],
    EXTRACT::PERFORMANCE_TRENDS[scores_over_time],
    CALCULATE::TRAJECTORIES[improving,stable,degrading],
    NOTE::SIGNIFICANT_EVENTS[version_changes,configuration_updates],
    ANALYZE::IMPLICATIONS[reliability_predictions,selection_guidance]
  ]
  OUTPUT::"Effectiveness timeline with trend analysis and evidence"

PATTERN_RECOGNITION:
  PURPOSE::"Identify patterns in test failures/successes"
  PROCESS::[
    ANALYZE::ACROSS_TESTS[multiple_test_results],
    IDENTIFY::COMMON_FAILURE_MODES[what_consistently_fails],
    RECOGNIZE::STRENGTHS[what_consistently_succeeds],
    CORRELATE::WITH_CONTEXT[test_conditions,model_versions,configurations],
    SYNTHESIZE::INSIGHTS[actionable_recommendations]
  ]
  OUTPUT::"Pattern analysis with supporting evidence citations"
